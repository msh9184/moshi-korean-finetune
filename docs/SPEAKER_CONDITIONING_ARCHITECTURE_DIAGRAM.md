```
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  3. REFERENCE SAMPLING                                                              │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  ReferenceSampler.sample()                                                   │  │
    │   │                                                                              │  │
    │   │  # 1. 유효 영역 계산 (avoid_overlap=True)                                    │  │
    │   │  valid_regions = [(0, exclude_start), (exclude_end, T)]                     │  │
    │   │                                                                              │  │
    │   │  # 2. 랜덤 duration 선택 (10-15초)                                           │  │
    │   │  duration_frames = randint(min_frames, max_frames)                          │  │
    │   │                                                                              │  │
    │   │  # 3. 시작 위치 샘플링                                                       │  │
    │   │  start_frame = sample_from_valid_regions()                                  │  │
    │   │                                                                              │  │
    │   │  # 4. Reference 추출                                                         │  │
    │   │  reference_codes = codes[:, start_frame:end_frame]                          │  │
    │   │  reference_audio = raw_audio[start_sample:end_sample]                       │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    │   Output: reference_codes [B, 9, T_ref], reference_audio [B, T_audio_ref]          │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  4. SPEAKER EMBEDDING EXTRACTION                                                    │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  speaker_encoder(reference_audio)                                            │  │
    │   │                                                                              │  │
    │   │  # 1. Resample 24kHz → 16kHz                                                 │  │
    │   │  resampled_audio = torchaudio.resample(reference_audio, 24000, 16000)       │  │
    │   │                                                                              │  │
    │   │  # 2. W2v-BERT 2.0 Forward                                                   │  │
    │   │  features = w2v_bert(resampled_audio, output_hidden_states=True)            │  │
    │   │                                                                              │  │
    │   │  # 3. Multi-layer Feature Aggregation                                        │  │
    │   │  mfa_features = concat(features.hidden_states[-25:], dim=-1)                │  │
    │   │                                                                              │  │
    │   │  # 4. Attentive Statistics Pooling                                           │  │
    │   │  pooled = ASP(mfa_features)  # → [B, 2*D]                                    │  │
    │   │                                                                              │  │
    │   │  # 5. Bottleneck                                                             │  │
    │   │  embedding = bottleneck(pooled)  # → [B, 256]                                │  │
    │   │                                                                              │  │
    │   │  # 6. L2 Normalize                                                           │  │
    │   │  speaker_embedding = F.normalize(embedding, p=2, dim=-1)                    │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    │   Output: speaker_embedding [B, 256]                                               │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  5. SPEAKER CONDITIONING                                                            │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  speaker_conditioner(speaker_embedding)                                      │  │
    │   │                                                                              │  │
    │   │  # Linear projection                                                         │  │
    │   │  projected = linear(speaker_embedding)  # [B, 256] → [B, 4096]              │  │
    │   │                                                                              │  │
    │   │  # Layer normalization (optional)                                            │  │
    │   │  normalized = layer_norm(projected)                                         │  │
    │   │                                                                              │  │
    │   │  # Learnable scale                                                           │  │
    │   │  sum_condition = normalized * scale  # scale=0.1 initially                   │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    │   Output: sum_condition [B, 4096]                                                  │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  6. AUDIO PROMPTING (VALL-E Style)                                                  │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  audio_prompt_module(codes, exclude_start, exclude_end)                      │  │
    │   │                                                                              │  │
    │   │  # 1. Sample prompt codes                                                    │  │
    │   │  prompt_samples = sampler.sample_batch(codes, exclude_start, exclude_end)   │  │
    │   │                                                                              │  │
    │   │  # 2. Apply prompts (prepend)                                                │  │
    │   │  prompted_codes, prompt_mask = sampler.apply_prompts(codes, prompt_samples) │  │
    │   │                                                                              │  │
    │   │  # prompted_codes: [B, 9, T_prompt + T_main]                                 │  │
    │   │  # prompt_mask:    [B, T_prompt + T_main]  - True for prompt positions      │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    │   Output: prompted_codes, prompt_mask                                              │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  7. MODEL FORWARD                                                                   │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  output = model(                                                             │  │
    │   │      codes=prompted_codes,                                                   │  │
    │   │      speaker_embedding=speaker_embedding,  # sum_condition 적용              │  │
    │   │  )                                                                            │  │
    │   │                                                                              │  │
    │   │  # Inside model.forward():                                                   │  │
    │   │  # 1. Embedding                                                              │  │
    │   │  hidden = embed(prompted_codes)  # [B, T_total, 4096]                        │  │
    │   │                                                                              │  │
    │   │  # 2. Add sum_condition (global speaker identity)                            │  │
    │   │  hidden = hidden + sum_condition.unsqueeze(1)                                │  │
    │   │                                                                              │  │
    │   │  # 3. Transformer layers (Prompt ↔ Main cross-attention 자연 발생)           │  │
    │   │  for layer in transformer_layers:                                            │  │
    │   │      hidden = layer(hidden)                                                  │  │
    │   │                                                                              │  │
    │   │  # 4. DepFormer output                                                       │  │
    │   │  text_logits = text_lm_head(hidden)                                          │  │
    │   │  audio_logits = depformer(hidden)                                            │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    │   Output: text_logits, audio_logits                                                │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  8. LOSS COMPUTATION                                                                │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌─────────────────────────────────────────────────────────────────────────────┐  │
    │   │  # Prompt 영역 제외한 Loss 계산                                               │  │
    │   │                                                                              │  │
    │   │  # prompt_mask 활용                                                          │  │
    │   │  valid_positions = ~prompt_mask  # Main sequence만                           │  │
    │   │                                                                              │  │
    │   │  text_loss = cross_entropy(                                                  │  │
    │   │      text_logits[:, valid_positions],                                        │  │
    │   │      text_targets[:, valid_positions]                                        │  │
    │   │  ) * text_padding_weight                                                     │  │
    │   │                                                                              │  │
    │   │  audio_loss = cross_entropy(                                                 │  │
    │   │      audio_logits[:, valid_positions],                                       │  │
    │   │      audio_targets[:, valid_positions],                                      │  │
    │   │      first_codebook_weight_multiplier=100.0                                  │  │
    │   │  )                                                                            │  │
    │   │                                                                              │  │
    │   │  total_loss = text_loss + audio_loss                                         │  │
    │   │                                                                              │  │
    │   └─────────────────────────────────────────────────────────────────────────────┘  │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  9. OPTIMIZATION                                                                    │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   loss.backward()                                                                  │
    │   optimizer.step()                                                                 │
    │   scheduler.step()                                                                 │
    │                                                                                    │
    │   → Repeat for all batches                                                         │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
```

```
    │   │   │                            │   │                                        ││ │
    │   │   │  Resample → W2v-BERT 2.0   │   │  Mimi Encode → Reference Codes         ││ │
    │   │   │  → speaker_embedding       │   │  → prompt_codes                         ││ │
    │   │   │  [1, 256]                  │   │  [1, 9, T_ref]                          ││ │

             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  2. AUTOREGRESSIVE GENERATION                                                       │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌──────────────────────────────────────────────────────────────────────────────┐ │
    │   │  Initial State                                                                │ │
    │   │                                                                               │ │
    │   │   sum_condition = speaker_conditioner(speaker_embedding)  # [1, 4096]        │ │
    │   │   current_codes = prompt_codes  # [1, 9, T_ref]                               │ │
    │   │                                                                               │ │
    │   └──────────────────────────────────────────────────────────────────────────────┘ │

    │   │       output = model(                                                         │ │
    │   │           codes=current_codes,                                                │ │
    │   │           speaker_embedding=speaker_embedding,                                │ │
    │   │       )                                                                        │ │

    │   │                                                                               │ │
    │   │       # 5. Update context                                                     │ │
    │   │       current_codes = concat(current_codes, [next_text, next_audio])         │ │
    │   │                                                                               │ │
    │   │       # 6. Check stopping condition                                           │ │
    │   │       if is_end_of_turn(next_text):                                           │ │
    │   │           break                                                               │ │
    │   │                                                                               │ │
    │   └──────────────────────────────────────────────────────────────────────────────┘ │
    │                                                                                    │
    └────────────────────────────────────────────────────────────────────────────────────┘
             │
             ↓
    ┌────────────────────────────────────────────────────────────────────────────────────┐
    │  3. OUTPUT                                                                          │
    ├────────────────────────────────────────────────────────────────────────────────────┤
    │                                                                                    │
    │   ┌──────────────────────────────────────────────────────────────────────────────┐ │
    │   │  Generated Audio (cloned voice)                                               │ │
```
